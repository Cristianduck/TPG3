import java.util.Scanner;

public class JuegoFutbol {
    public static void main(String[] args) {
        Scanner teclado = new Scanner(System.in);
        System.out.println("Digite el nombre del equipo que va a ganar");
        String equipoUno = teclado.nextLine();
        System.out.println("Digite el nombre del segundo equipo que va a ganar");
        String equipoDos = teclado.nextLine();

        int puntajeUno = 0, puntajeDos = 0;

        int partido = 1;
        while (partido<=8){
            int golesUno = (int)(Math.random()*5);
            int golesDos = (int)(Math.random()*5);
            if (golesUno>golesDos){
                puntajeUno += 3;
            }else if (golesDos>golesUno){
                puntajeDos += 3;
            }else if (golesUno> 0 && golesUno==golesDos) {
                puntajeUno++;
                puntajeDos++;
            }else {
                System.out.println("partido no valido");
                continue;

            }
            partido++;
            System.out.println("resultados del partido" +partido);
            System.out.println(equipoUno+"<"+golesUno+"-"+golesDos+">"+equipoDos+"" );
            if (puntajeUno>=15 || puntajeDos>=15){
                break;

            }

        }
        partido--;
        System.out.println("Se jugareon "+partido+" partidos.");
        System.out.println("el "+equipoUno+"consigio "+puntajeUno+"puntos");
        System.out.println("el "+equipoDos+"consigio "+puntajeDos+"puntos");
        if (puntajeUno>puntajeDos){
            System.out.println("felicitaciones tu equipo gano");
        }else {
            System.out.println("lo sentimos tu equipo no gano el campeonato");

        }

    }
}
